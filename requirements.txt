def load_json(path, default=None):
    if default is None:
        default = {}
    try:
        if path.exists():
            with open(path, "r", encoding="utf-8") as f:
                return json.load(f)
    except Exception:
        pass
    return dict(default) if isinstance(default, dict) else default


def save_json(path, data):
    try:
        with open(path, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2, default=str)
    except Exception:
        pass

def calc_tdee(p):
    w = float(p.get("weight", 80))
    h = float(p.get("height", 178))
    a = int(p.get("age", 28))
    if p.get("gender") == "Erkek":
        bmr = 10 * w + 6.25 * h - 5 * a + 5
    else:
        bmr = 10 * w + 6.25 * h - 5 * a - 161
    mult = {
        "Hareketsiz": 1.2, "Az Hareketli": 1.375,
        "Orta Düzey": 1.55, "Aktif": 1.725, "Çok Aktif": 1.9,
    }
    tdee = bmr * mult.get(p.get("activity", "Orta Düzey"), 1.55)
    goal = p.get("goal", "Kilo Koruma")
    if goal == "Kilo Verme":
        tdee -= 500
    elif goal == "Kas Yapma":
        tdee += 300
    return round(tdee)


def get_bmi(p):
    h = float(p.get("height", 178)) / 100
    w = float(p.get("weight", 80))
    return round(w / (h * h), 1) if h > 0 else 0


def bmi_cat(b):
    if b < 18.5:
        return "Zayıf", "⚠️"
    if b < 25:
        return "Normal", "✅"
    if b < 30:
        return "Fazla Kilolu", "⚠️"
    return "Obez", "🔴"

def ask_gemini(prompt, system=""):
    if not API_KEY:
        return "⚠️ API anahtarı bulunamadı. Streamlit ayarlarından GEMINI_API_KEY ekleyin."
    try:
        genai.configure(api_key=API_KEY)
        model = genai.GenerativeModel(
            "gemini-2.0-flash",
            system_instruction=system if system else None,
        )
        resp = model.generate_content(prompt)
        return resp.text
    except Exception as e:
        return "❌ Gemini Hatası: " + str(e)


def _parse_json(text):
    text = text.strip()
    # 1 — doğrudan
    try:
        return json.loads(text)
    except Exception:
        pass
    # 2 — code block
    m = re.search(r"```(?:json)?\s*\n?(.*?)```", text, re.DOTALL)
    if m:
        try:
            return json.loads(m.group(1).strip())
        except Exception:
            pass
    # 3 — ilk {...}
    m = re.search(r"\{[^{}]*\}", text, re.DOTALL)
    if m:
        try:
            return json.loads(m.group(0))
        except Exception:
            pass
    # 4 — ilk { son }
    s, e = text.find("{"), text.rfind("}")
    if s != -1 and e != -1 and e > s:
        try:
            return json.loads(text[s:e + 1])
        except Exception:
            pass
    return None


def estimate_calories(food):
    if not API_KEY:
        return None, "API anahtarı bulunamadı."
    try:
        genai.configure(api_key=API_KEY)
        model = genai.GenerativeModel(
            "gemini-2.0-flash",
            system_instruction="Sen bir beslenme uzmanısın. Türk mutfağını iyi bilirsin. Sadece JSON formatında yanıt ver.",
            generation_config=genai.GenerationConfig(
                response_mime_type="application/json",
            ),
        )
        prompt = (
            "Bu yemeğin besin değerlerini JSON olarak ver.\n"
            'Format: {"food":"yemek adı","calories":sayı,"protein":sayı,"carbs":sayı,"fat":sayı,"portion":"porsiyon"}\n\n'
            "Yemek: " + food
        )
        resp = model.generate_content(prompt)
        raw = resp.text.strip()
        data = _parse_json(raw)
        if data and isinstance(data, dict) and data.get("calories", 0) > 0:
            return data, None
        return None, "Parse edilemedi: " + raw[:200]
    except Exception as e:
        return None, str(e)

            pv = body[sd[1]] if len(sd) >= 2 else None
            pd2 = sd[1] if len(sd) >= 2 else None

            ma = (datetime.now() - timedelta(days=30)).strftime("%Y-%m-%d")
            mv, md2 = None, None
            for d in sorted(body.keys()):
                if d <= ma:
                    md2, mv = d, body[d]

            specs = [
                ("⚖️ Kilo", "weight", "kg", True),
                ("🔴 Yağ", "fat", "%", True),
                ("💪 Kas", "muscle", "%", False),
                ("💧 Su", "water", "%", False),
            ]

            def show_changes(cmp, cmp_d, title):
                if not cmp:
                    return
                st.markdown(f"<div style='color:var(--t3);font-size:.75rem;font-weight:600;margin:12px 0 6px;text-transform:uppercase;letter-spacing:1px'>{title} ({cmp_d})</div>", unsafe_allow_html=True)

                    ok = (ch < 0) if rev else (ch > 0)

                        cls, clr, sts = "cf", "var(--acc2)", "Değişim yok"
                    elif ok:
                        cls, clr, sts = "cu", "var(--grn)", "Gelişme ✓"
                    else:
                        cls, clr, sts = "cd", "var(--red)", "Gerileme ✗"
                    ar = "↗" if ch > 0 else ("↘" if ch < 0 else "→")
                    st.markdown(f'<div class="cc {cls}"><div><div style="color:var(--t3);font-size:.7rem">{lb}</div><div style="color:var(--t1);font-size:1.15rem;font-weight:700">{cu}{un}</div></div><div style="text-align:right"><div style="color:{clr};font-size:1.1rem;font-weight:700">{ar} %{abs(round(ch, 1))}</div><div style="color:{clr};font-size:.65rem">{sts}</div></div></div>', unsafe_allow_html=True)

            st.markdown(f"<div style='color:var(--acc2);font-size:.8rem;font-weight:600'>Son: {ld}</div>", unsafe_allow_html=True)
            show_changes(pv, pd2, "Önceki ölçüme göre")
            show_changes(mv, md2, "Aylık değişim")

            if not pv and not mv:
                st.info("Karşılaştırma için en az 2 ölçüm gerekli.")
        else:
            st.markdown('<div class="glass" style="text-align:center;padding:40px"><div style="font-size:2.5rem">📏</div><div style="color:var(--t2);margin-top:8px">İlk ölçümünü ekle</div></div>', unsafe_allow_html=True)

        st.markdown("<div style='height:20px'></div>", unsafe_allow_html=True)
        st.markdown('<div class="st">📈 Trend</div>', unsafe_allow_html=True)
        sds = sorted(body.keys())
        df = pd.DataFrame([
            {"Tarih": d, "Kilo": body[d].get("weight", 0), "Yağ %": body[d].get("fat", 0), "Kas %": body[d].get("muscle", 0)}
            for d in sds
        ]).set_index("Tarih")
        g1, g2 = st.columns(2)
        with g1:
            st.caption("⚖️ Kilo")
            st.line_chart(df[["Kilo"]], color="#8b5cf6")
        with g2:
            st.caption("🔴 Yağ / 💪 Kas")
            st.line_chart(df[["Yağ %", "Kas %"]])
